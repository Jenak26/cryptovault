\section{Guiding Principles}
% =====================================================================
\begin{enumerate}[leftmargin=*]
  \item \textbf{Vertical slices, not horizontal.} Get one feature working end-to-end (DB $\to$ service $\to$ API) before starting the next.
  \item \textbf{Backend first, frontend last.} Build React only once the API is stable.
  \item \textbf{Own your schema} with Flyway migrations — never \code{ddl-auto=update} in anything real.
  \item \textbf{Crypto discipline:} fresh random nonce every encryption; never log keys or plaintext; never hard-delete a referenced key/algorithm.
  \item \textbf{Commit small and often.} One logical change per commit.
  \item \textbf{Every phase ends with a test} proving it works.
\end{enumerate}

\section{Implementation Phases}
% =====================================================================
Each phase: \textbf{Goal $\to$ Tasks $\to$ Done when}. Estimates are beginner-realistic
(roughly double the original, since Java/Spring is new to you).

\subsection{Phase 0 — Environment \& Skeleton ($\sim$1 week)}
\textbf{Goal:} A running, empty Spring Boot app talking to MySQL + Redis.
\begin{itemize}[leftmargin=*]
  \item Install JDK 21, IntelliJ IDEA Community, Docker Desktop, Git.
  \item Generate project at \href{https://start.spring.io}{start.spring.io}: Web, Security, Data JPA, Data Redis, Validation, MySQL Driver, Flyway, Lombok.
  \item Add \code{jjwt} and \code{bcprov-jdk18on} (Bouncy Castle).
  \item Write \code{docker-compose.yml} (MySQL 8 + Redis 7); configure \code{application.yml}.
  \item Add \code{GET /api/health} returning \code{200}.
\end{itemize}
\textbf{Done when:} \code{docker compose up} runs MySQL+Redis, app boots, \code{/api/health} returns 200.

\subsection{Phase 1 — Database Schema ($\sim$3–4 days)}
\textbf{Goal:} All five tables exist, versioned and seeded.
\begin{itemize}[leftmargin=*]
  \item JPA entities: Role, User, CryptoKey, VaultRecord, AuditLog.
  \item Flyway \code{V1\_\_initial\_schema.sql} (full DDL + indexes); \code{V2\_\_seed\_roles.sql}.
  \item \code{ddl-auto=validate} so Hibernate checks but never edits the schema.
\end{itemize}
\textbf{Done when:} Flyway applies V1+V2, \code{roles} has two rows, entities validate.

\subsection{Phase 2 — Authentication Core ($\sim$1.5 weeks)}
\textbf{Goal:} Register, login, logout with hashed passwords and JWTs.
\begin{itemize}[leftmargin=*]
  \item \code{AuthService}: register (bcrypt + issue JWT), login (verify + issue JWT), logout (blacklist \code{jti} in Redis).
  \item \code{JwtService} (jjwt): create/validate tokens.
  \item \code{AuthController}: register / login / logout. DTOs as records with validation.
\end{itemize}
\textbf{Done when:} Register $\to$ JWT; login $\to$ JWT; logout $\to$ token rejected; wrong password $\to$ generic 401.

\subsection{Phase 3 — Security Filter \& RBAC ($\sim$1 week)}
\textbf{Goal:} Protected endpoints; admin-only routes enforced.
\begin{itemize}[leftmargin=*]
  \item \code{JwtAuthFilter}: validate token, check blacklist, set security context.
  \item \code{SecurityConfig}: stateless, public auth routes, rest authenticated; \code{@PreAuthorize}.
  \item \code{RateLimiter} (Redis) on login.
\end{itemize}
\textbf{Done when:} No token $\to$ 401; user on admin route $\to$ 403; admin $\to$ 200; rapid bad logins rate-limited.

\subsection{Phase 4 — Crypto Engine (crypto-agility core) ($\sim$1.5 weeks)}
\textbf{Goal:} Pluggable encryption selected by config; two algorithms.
\begin{itemize}[leftmargin=*]
  \item \code{CipherStrategy} interface; \code{AesGcmStrategy} (fresh 12-byte nonce/call); \code{ChaCha20Poly1305Strategy}.
  \item \code{CryptoEngine}: encrypt with active algorithm, decrypt with the algorithm named in the blob.
  \item Unit tests: round-trip both; tampering fails (auth tag).
\end{itemize}
\textbf{Done when:} Switching \code{AES}$\to$\code{CHACHA20} in config changes new writes, no code change; tampered ciphertext throws.
\begin{note}\textbf{The one rule:} a unique random nonce for every encryption.\end{note}

\subsection{Phase 5 — Key Management \& Rotation ($\sim$1 week)}
\textbf{Goal:} Versioned keys; rotate without breaking old data.
\begin{itemize}[leftmargin=*]
  \item \code{KeyManager}: active key; key material encrypted under a master secret from an env var.
  \item \code{POST /api/admin/rotate-key}: new version active, previous retired.
\end{itemize}
\textbf{Done when:} Rotate $\to$ new writes use v2; old v1 records still decrypt; v1 marked retired, never deleted.

\subsection{Phase 6 — Vault Service ($\sim$1 week)}
\textbf{Goal:} Store and retrieve encrypted records end-to-end.
\begin{itemize}[leftmargin=*]
  \item \code{store}: encrypt (active algo+key), save row with algorithm + key\_version + user.
  \item \code{retrieve}: read row's algorithm + key\_version, fetch key, decrypt. Soft-delete.
\end{itemize}
\textbf{Done when:} Switch active algorithm, store a second secret — both retrieve correctly.

\subsection{Phase 7 — Audit Logging ($\sim$3–4 days)}
\textbf{Goal:} Every security-relevant action is logged (nullable user).
\textbf{Done when:} Failed login on an unknown email logs a row with \code{user\_id = NULL};
\code{GET /api/admin/audit} is paginated and filterable.

\subsection{Phase 8 — Frontend (React + TypeScript) ($\sim$2 weeks)}
\textbf{Goal:} Banking-style UI over the stable API: Landing, Login/Register, Dashboard, Vault, Admin (charts).\\
\textbf{Done when:} Full browser flow: register $\to$ login $\to$ store $\to$ vault table $\to$ admin charts $\to$ logout.

\subsection{Phase 9 — Reassess before V2/V3 (stretch)}
V2: MFA (TOTP / email OTP), failed-login alerts, idempotent re-encryption jobs.
V3: Post-quantum via Bouncy Castle — ML-KEM, ML-DSA, hybrid \code{X25519 + ML-KEM}.

\section{Testing, Risks, Definition of Done}
% =====================================================================
\subsection{Testing strategy}
\begin{itemize}[leftmargin=*]
  \item \textbf{Unit (JUnit 5 + Mockito):} crypto round-trips, tamper detection, JWT validity, hashing, rate limiter.
  \item \textbf{Integration (\code{@SpringBootTest} + Testcontainers):} real MySQL/Redis; full register$\to$login$\to$store$\to$retrieve; rotation.
  \item \textbf{Manual (Postman):} every endpoint + edge cases.
  \item \textbf{Security:} passwords/keys/plaintext never in logs or responses; generic auth errors.
\end{itemize}

\subsection{Risk register}
\renewcommand{\arraystretch}{1.25}
\begin{longtable}{p{5cm} p{2cm} p{6cm}}
\hline
\textbf{Risk} & \textbf{Likelihood} & \textbf{Mitigation} \\ \hline
Spring/Java learning curve stalls progress & High & Do a Spring Boot + MySQL tutorial first; budget 2$\times$ time. \\
Docker/MySQL setup friction & Medium & Use provided \code{docker-compose.yml}; don't install MySQL natively. \\
Nonce reuse / crypto mistake & Medium & Fresh random nonce per call; unit-test tamper detection. \\
Scope creep into V2/V3 too early & High & Hard stop after Phase 8; post-quantum is a stretch goal. \\
Key material mishandled & Medium & Master key from env var; never store raw keys in DB plaintext. \\ \hline
\end{longtable}

\subsection{Definition of Done (MVP)}
\begin{itemize}[leftmargin=*]
  \item \code{docker compose up} brings up MySQL + Redis; app boots clean.
  \item Flyway-managed schema, five tables, roles seeded.
  \item Register / login / logout with bcrypt + JWT + Redis blacklist.
  \item JWT filter + RBAC enforced; login rate-limited.
  \item Two algorithms behind \code{CipherStrategy}; active one config-switchable.
  \item Store/retrieve; old records decrypt after algorithm switch; rotation safe.
  \item Full audit trail incl. nullable-user failed logins.
  \item React UI: landing, auth, dashboard, vault, admin charts.
  \item Unit + integration tests green; README with run instructions.
\end{itemize}
